      #include <FastLED.h>

      #define LEDpin 12
      #define NUM_LEDS 24
      #define LED_TYPE WS2812
      #define COLOR_ORDER GRB
      
      #define Apin 13
      #define ANALOG A0
      int x = 0;
      uint8_t max_bright = 128;
      CRGB leds[NUM_LEDS];
      
      void setup(){
        Serial.begin(115200);
        pinMode(A0, INPUT);
        pinMode(Apin, OUTPUT);
        LEDS.addLeds<LED_TYPE,LEDpin,COLOR_ORDER>(leds,NUM_LEDS);
        FastLED.setBrightness(max_bright);
      }
      
      int loop_index = 1;
      
      void loop(){
        digitalWrite(Apin, HIGH);
        x = digitalRead(ANALOG);
        Serial.println(x);
        
        if(x == 1){
          fill_gradient(leds, 0, CHSV(50, 255,255) , NUM_LEDS, CHSV(150,255,255), LONGEST_HUES); 
          //shutdown led before index
          for(int i=0; i<loop_index - 3; i++){
            leds[i] = CRGB::Black;
          }
      
          //shutdown led after index
          if(loop_index < NUM_LEDS){
            for(int j=loop_index; j<NUM_LEDS; j++){
              leds[j] = CRGB::Black;
            }
          }
          loop_index++;
        }else{
          loop_index = 1;
          for(int i=0; i<NUM_LEDS; i++){
            leds[i] = CRGB::Black;
          }
        }
        FastLED.show();
        delay(300);
        
        if(loop_index >= NUM_LEDS){
          loop_index = 1;
        }
      }
